题解 CF1352B 【Same Parity Summands】

2020-05-29 4 min read # 数学 # 数论

题目传送门1

题目传送门2

这是史上第一次Div.4!

话说好像好水的样子啊

说正事：

题意：求出能否将nnn分成kkk个奇偶性相同的正整数，如果能，输出任意一种方法。

按nnn与kkk的奇偶性分类：

  • 当n≡1(mod2)n\equiv1\pmod{2}n≡1(mod2)且k≡0(mod2)k\equiv0\pmod{2}k≡0(mod2)时，不管是kkk个奇数还是kkk个偶数和都是偶数。所以输出−1-1−1。

  • 当n≡0(mod2)n\equiv0\pmod{2}n≡0(mod2)且k≡0(mod2)k\equiv0\pmod{2}k≡0(mod2)时，都是奇数或都是偶数都可以。因为奇数也就是全是111最小，所以取奇数。如果n<kn<kn<k，则无论如何满足不了，输出−1-1−1，否则先把kkk个数都设为111，这是和一定是偶数，每次如果不够就在最后加222，直到和为kkk时为止。其实就可以先把前(k−1)(k-1)(k−1)个数设成111，最后一个数为(n−k+1)(n-k+1)(n−k+1)。

  • 当n≡1(mod2)n\equiv1\pmod{2}n≡1(mod2)且k≡1(mod2)k\equiv1\pmod{2}k≡1(mod2)时，这时必须所有数取奇数。同样的，如果n<kn<kn<k，则无论如何满足不了，输出−1-1−1，否则先把前(k−1)(k-1)(k−1)个数设成111，最后一个数为(n−k+1)(n-k+1)(n−k+1)。

  • 当n≡0(mod2)n\equiv0\pmod{2}n≡0(mod2)且k≡1(mod2)k\equiv1\pmod{2}k≡1(mod2)时，此时必须所有数取偶数，如果n<2×kn<2\times kn<2×k，则无论如何满足不了，输出−1-1−1，否则先把前(k−1)(k-1)(k−1)个数设成222，最后一个数为(n−2×k+2)(n-2\times k+2)(n−2×k+2)。

说完了，上代码：

#include<stdio.h>
#include<algorithm>
using namespace std;
void read(int &x)
{
	x=0;
	int bei=1;
	char ch=getchar();
	while(ch<'0'||ch>'9')
	{
		if(ch=='-')
			bei=-1;
		ch=getchar();
	}
	while(ch>='0'&&ch<='9')
	{
		x=x*10+ch-'0';
		ch=getchar();
	}
	x*=bei;
}
int main()
{
	int t,n,k;
	read(t);
	while(t--)
	{
		read(n),read(k);
		if(n%2==1&&k%2==0)
		{
			puts("NO");
			continue;
		}
		if(n%2==1&&n<k)
		{
			puts("NO");
			continue;
		}
		if(n%2==0&&k%2==0&&n<k)
		{
			puts("NO");
			continue;
		}
		if(n%2==0&&k%2==1&&n<k*2)
		{
			puts("NO");
			continue;
		}
		puts("YES");
		if(n%2==0&&k%2==1)
		{
			for(int i=1;i<k;++i)
				printf("2 ");
			printf("%d\n",n-2*(k-1));
		}
		else
		{
			for(int i=1;i<k;++i)
				printf("1 ");
			printf("%d\n",n-(k-1));
		}
	}
	return 0;
}
